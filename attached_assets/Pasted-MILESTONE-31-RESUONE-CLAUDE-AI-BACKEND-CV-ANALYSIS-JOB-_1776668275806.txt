MILESTONE 31 — RESUONE + CLAUDE AI BACKEND (CV ANALYSIS + JOB MATCH + RECRUITER SUMMARY)

GOAL
Build the first Claude-powered AI layer for Resuone so users can:
1. Upload CVs (PDF/DOCX/TXT)
2. Paste a job description
3. Receive:
   - parsed candidate profile
   - ATS-style match score
   - strengths
   - gaps
   - recommended improvements
   - recruiter summary
4. Save all results in Supabase
5. Use streaming responses in the UI for better UX
6. Keep all Anthropic API calls on the backend only

IMPORTANT RULES
- NEVER call Anthropic directly from frontend
- NEVER expose ANTHROPIC_API_KEY to browser
- All AI analysis must go through backend API routes
- Save every analysis result in database
- Keep raw extraction separate from normalized structured output
- Add rate limits on analysis routes
- Add usage logging for every Claude request
- Make the system modular so later I can swap Claude / OpenAI / Gemini if needed

TECH STACK
- Frontend: Next.js App Router
- Backend: Next.js route handlers
- Database: Supabase Postgres
- Storage: Supabase Storage
- Auth: existing Resuone auth flow
- AI: Anthropic Claude via official SDK
- Validation: Zod
- File parsing:
  - PDF text extraction
  - DOCX text extraction
  - plain text fallback
- Streaming: server-side streaming from Claude to frontend
- Queue-ready structure for future bulk analysis

import pdfParse from "pdf-parse";
import mammoth from "mammoth";

export async function extractTextFromFile(file: File): Promise<string> {
  const buffer = Buffer.from(await file.arrayBuffer());
  const name = file.name.toLowerCase();

  if (name.endsWith(".pdf")) {
    const parsed = await pdfParse(buffer);
    return parsed.text || "";
  }

  if (name.endsWith(".docx")) {
    const result = await mammoth.extractRawText({ buffer });
    return result.value || "";
  }

  if (name.endsWith(".txt")) {
    return buffer.toString("utf-8");
  }

  throw new Error("Unsupported file type. Please upload PDF, DOCX, or TXT.");
}

CREATE lib/utils/rate-limit.ts

const requestMap = new Map<string, { count: number; ts: number }>();

export function basicRateLimit(key: string, limit = 10, windowMs = 60_000) {
  const now = Date.now();
  const existing = requestMap.get(key);

  if (!existing || now - existing.ts > windowMs) {
    requestMap.set(key, { count: 1, ts: now });
    return true;
  }

  if (existing.count >= limit) {
    return false;
  }

  existing.count += 1;
  requestMap.set(key, existing);
  return true;
}

CREATE app/api/ai/analyze-cv/route.ts
Paste:

import { NextRequest, NextResponse } from "next/server";
import { anthropic, AI_MODELS } from "@/lib/ai/anthropic";
import { buildCVParsingPrompt } from "@/lib/ai/prompts";
import { ParsedCVSchema } from "@/lib/ai/schemas";
import { extractTextFromFile } from "@/lib/files/extract-text";
import { supabaseAdmin } from "@/lib/supabase/admin";
import { basicRateLimit } from "@/lib/utils/rate-limit";
import { logAIUsage } from "@/lib/utils/usage-log";

export async function POST(req: NextRequest) {
  const ip = req.headers.get("x-forwarded-for") || "unknown";
  if (!basicRateLimit(`analyze-cv:${ip}`, 8, 60_000)) {
    return NextResponse.json({ error: "Rate limit exceeded" }, { status: 429 });
  }

  try {
    const formData = await req.formData();
    const file = formData.get("file") as File | null;
    const userId = String(formData.get("userId") || "");

    if (!file) {
      return NextResponse.json({ error: "File is required" }, { status: 400 });
    }

    const extractedText = await extractTextFromFile(file);

    const completion = await anthropic.messages.create({
      model: AI_MODELS.MAIN,
      max_tokens: 3000,
      temperature: 0.2,
      messages: [
        {
          role: "user",
          content: buildCVParsingPrompt(extractedText)
        }
      ]
    });

    const text = completion.content
      .filter((c: any) => c.type === "text")
      .map((c: any) => c.text)
      .join("");

    const parsed = ParsedCVSchema.parse(JSON.parse(text));

    const docInsert = await supabaseAdmin
      .from("candidate_documents")
      .insert({
        user_id: userId,
        file_name: file.name,
        file_type: file.type,
        extracted_text: extractedText
      })
      .select("id")
      .single();

    const analysisInsert = await supabaseAdmin
      .from("analyses")
      .insert({
        user_id: userId,
        candidate_document_id: docInsert.data?.id ?? null,
        analysis_type: "cv_parse",
        model_name: AI_MODELS.MAIN,
        input_tokens: completion.usage?.input_tokens ?? 0,
        output_tokens: completion.usage?.output_tokens ?? 0,
        result_json: parsed
      })
      .select("id")
      .single();

    await logAIUsage({
      userId,
      route: "/api/ai/analyze-cv",
      modelName: AI_MODELS.MAIN,
      inputTokens: completion.usage?.input_tokens ?? 0,
      outputTokens: completion.usage?.output_tokens ?? 0,
      success: true
    });

    return NextResponse.json({
      ok: true,
      candidateDocumentId: docInsert.data?.id,
      analysisId: analysisInsert.data?.id,
      extractedText,
      parsed
    });
  } catch (error: any) {
    await logAIUsage({
      route: "/api/ai/analyze-cv",
      success: false,
      errorMessage: error.message
    });

    return NextResponse.json(
      { error: error.message || "Failed to analyze CV" },
      { status: 500 }
    );
  }
}

CREATE app/api/ai/stream-analysis/route.ts
Paste:

import { NextRequest } from "next/server";
import { anthropic, AI_MODELS } from "@/lib/ai/anthropic";

export async function POST(req: NextRequest) {
  const body = await req.json();
  const { prompt } = body;

  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    async start(controller) {
      try {
        const claudeStream = await anthropic.messages.stream({
          model: AI_MODELS.MAIN,
          max_tokens: 2500,
          temperature: 0.3,
          messages: [{ role: "user", content: prompt }]
        });

        for await (const text of claudeStream.textStream) {
          controller.enqueue(encoder.encode(text));
        }

        controller.close();
      } catch (err: any) {
        controller.enqueue(encoder.encode(`ERROR: ${err.message}`));
        controller.close();
      }
    }
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/plain; charset=utf-8",
      "Cache-Control": "no-cache, no-transform",
    },
  });
}

"use client";

import { useState } from "react";

export default function AnalysisPage() {
  const [file, setFile] = useState<File | null>(null);
  const [jobText, setJobText] = useState("");
  const [parsedCV, setParsedCV] = useState<any>(null);
  const [jobResult, setJobResult] = useState<any>(null);
  const [loading, setLoading] = useState(false);

  const userId = "TEMP_USER_ID";

  async function handleParseCV() {
    if (!file) return;
    setLoading(true);

    const fd = new FormData();
    fd.append("file", file);
    fd.append("userId", userId);

    const res = await fetch("/api/ai/analyze-cv", {
      method: "POST",
      body: fd
    });

    const data = await res.json();
    setParsedCV(data.parsed || null);
    setLoading(false);
  }

  async function handleMatch() {
    if (!parsedCV || !jobText) return;
    setLoading(true);

    const res = await fetch("/api/ai/analyze-job-match", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        userId,
        parsedCV,
        jobText
      })
    });

    const data = await res.json();
    setJobResult(data.result || null);
    setLoading(false);
  }

  return (
    <div className="p-6 max-w-5xl mx-auto space-y-6">
      <h1 className="text-2xl font-bold">Resuone AI Analysis</h1>

      <div className="space-y-2">
        <label className="block font-medium">Upload CV</label>
        <input
          type="file"
          accept=".pdf,.docx,.txt"
          onChange={(e) => setFile(e.target.files?.[0] || null)}
        />
        <button
          className="px-4 py-2 rounded bg-black text-white"
          onClick={handleParseCV}
          disabled={!file || loading}
        >
          {loading ? "Parsing..." : "Parse CV"}
        </button>
      </div>

      {parsedCV && (
        <pre className="p-4 rounded border overflow-auto text-sm">
          {JSON.stringify(parsedCV, null, 2)}
        </pre>
      )}

      <div className="space-y-2">
        <label className="block font-medium">Paste Job Description</label>
        <textarea
          className="w-full min-h-[220px] border rounded p-3"
          value={jobText}
          onChange={(e) => setJobText(e.target.value)}
          placeholder="Paste full job description here..."
        />
        <button
          className="px-4 py-2 rounded bg-black text-white"
          onClick={handleMatch}
          disabled={!parsedCV || !jobText || loading}
        >
          {loading ? "Analyzing..." : "Analyze Match"}
        </button>
      </div>

      {jobResult && (
        <pre className="p-4 rounded border overflow-auto text-sm">
          {JSON.stringify(jobResult, null, 2)}
        </pre>
      )}
    </div>
  );
}
